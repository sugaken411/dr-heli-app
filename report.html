// 🌟 データベースからの同期と【手入力データの保護マージ ＋ 重複アラート】 async function syncFromDb(force) { showLoader(true); const dateVal = document.getElementById('info-date').value; try { const res = await fetch(GAS_API_URL, { method: "POST", headers: { "Content-Type": "text/plain;charset=utf-8" }, body: JSON.stringify({ action: "fetch_daily_report", password: PIN_CODE, date: dateVal, force_sync: force }) }); const json = await res.json(); if (json.status === "success") { const data = json.data; if (force) { // 【再同期】時間の重複チェック const manualEvents = events.filter(e => !String(e.id).startsWith("auto-")); const autoEvents = (data.events || []).filter(e => String(e.id).startsWith("auto-")); let hasOverlap = false; for (let me of manualEvents) { const ms = timeToMins(me.start); const me_end = timeToMins(me.end); for (let ae of autoEvents) { const as = timeToMins(ae.start); const ae_end = timeToMins(ae.end); // 時間が被っているか判定 if (ms < ae_end && me_end > as) { hasOverlap = true; break; } } if(hasOverlap) break; } if (hasOverlap) { if(!confirm("⚠️ 取り込もうとしている事案の活動時間が、すでに入力されている日誌の活動時間と重複しています。\n\n事案データで上書きして取り込みますか？\n（※活動時間自体の修正は「事案検索コンソール」から行ってください）")) { showLoader(false); return; } } else { if(!confirm("事案データベースから最新の症例登録情報を取り込みますか？")) { showLoader(false); return; } } events = [...manualEvents, ...autoEvents]; alert("事案データを取り込みました。"); } else { // 【初回読込】 const setIfNotSelected = (selectId, otherId, val) => { const sel = document.getElementById(selectId); if (val && (!sel.value || sel.value === '')) { if ([...sel.options].some(opt => opt.value === val)) { sel.value = val; } else { sel.value = 'その他'; document.getElementById(otherId).value = val; document.getElementById(otherId).classList.remove('hidden'); } } }; setIfNotSelected('info-nurse', 'info-nurse-other', data.recorder); setIfNotSelected('info-doctor', 'info-doctor-other', data.doctor); document.querySelectorAll('input[name="reason"]').forEach(cb => cb.checked = false); if (data.hamamachiReason) { const reasons = data.hamamachiReason.split(','); reasons.forEach(r => { const cb = document.querySelector(`input[name="reason"][value="${r.trim()}"]`); if(cb) cb.checked = true; }); } events = data.events || []; } updatePrintHeader(); render(); } } catch(e) { alert("データの同期に失敗しました。"); } finally { showLoader(false); } }
